VH VV VP VR ; PH PV PP PR ; RH RV RP RR        H = HORIZONTAL POLARIZED PARALLEL TO REFERENCE FRAME;  V = VERTICAL POLARIZED PERPENDICULAR TO REFERENCE FRAME;  P = 45 POLARIZED;  R = RIGHT CIRCULAR POLARIZED;    /*************//* Examples  /*************/sphere diameter = 2.00000;
